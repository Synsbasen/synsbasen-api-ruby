def self.configure
  @config ||= OpenStruct.new
  yield(@config) if block_given?

  raise "Missing configuration. Required configurations are #{REQUIRED_CONFIGS}" unless REQUIRED_CONFIGS.all? { |c| @config[c] }

  @config
end
